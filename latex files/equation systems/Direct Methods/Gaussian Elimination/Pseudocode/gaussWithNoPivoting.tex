\begin{algorithmic}[1]
\Procedure{Gauss no pivoting}{}
 \For{k=1,...,n-1}
  \State PIVOTING: Skip
  \State ELIMINATION
  \If{($a_kk\neq0$)}
    \For{i=k+1,...,n}
      \State $a_ik <- l_ik = \frac{a_ik}{a_kk}$

     \EndFor
      \For{i=k+1,...,n}
        \For{j=k+1,...,n}
        \State $a_ij <- a_ij - l_ik a_kj$
        \EndFor
       \EndFor 
  \Else
   \State Stop, algorith failks when $a_kk=0$
  \EndIf 
  \EndFor
\EndProcedure
