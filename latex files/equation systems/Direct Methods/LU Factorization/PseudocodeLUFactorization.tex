\caption{LU Factorization}\label{numerical methods}
\begin{algorithmic}[1]
\Procedure{LU Factorization}{}
 \For{k=1,...,n-1}
  \For{i=k+1:n}
    \State $a(i,k)=a(i,k)/a(k,k)$
    \For{j=k+1:n}
      \State a(i,j)=a(i,j)-a(i,k).a(k,j)
     \EndFor
    \EndFor    
  \EndFor

  \State $x(1)=b(1)$
  \For{i=2:n}
     \State s=0
     \For{j=1:i-1}
       \State s=s+a(i,j)*x(j)
      \EndFor
     \State x(i)=b(i)-s
   \EndFor  
\EndProcedure
\end{algorithmic}
\end{algorithm}
